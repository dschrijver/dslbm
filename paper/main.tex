\section{Knudsen expansion}
In diffusive scaling, we get the following for a Knudsen expansion,

\begin{equation}
\begin{split}
    f_i^{(0)} &= f_i^{\text{eq}, (0)} = w_i \rho^{(0)},\\
    f_i^{(1)} &= f_i^{\text{eq}, (1)} = w_i \rho^{(1)} + w_i c_{i\alpha} \rho^{(0)} \frac{ u^{(1)}_\alpha }{c_s^2},\\
    f_i^{(2)} &= f_i^{\text{eq}, (2)} - \frac{\tau}{\textit{Sh}}c_{i\alpha}\partial_\alpha f_i^{\text{eq},(1)},\\
    f_i^{(3)} &= f_i^{\text{eq}, (3)} - \tau \partial_t f_i^{\text{eq}, (1)} - \frac{\tau}{\textit{Sh}}c_{i\alpha}\partial_\alpha f_i^{\text{eq},(2)}\\
    &\quad + \frac{\tau}{\textit{Sh}^2}c_{i\alpha}\partial_\alpha\Biggl(\Bigl(\tau - \frac{\Delta t}{2}\Bigr)c_{i\beta}\partial_\beta f_i^{\text{eq},(1)}\Biggr)
\end{split}
\end{equation}
where repeated Greek indices are implicitly summed over. At the boundary, the outgoing and parallel populations are governed by the LBE, which means that they are given by the equation above. The incoming populations are given by the NEBB,
\begin{equation}
\begin{split}
    f_i &= f_q + 2w_i \rho \frac{\bm{u}\cdot \bm{c}_i}{c_s^2} - \sum_\alpha w_i c_{i\alpha} N_\alpha,\\
    N_\alpha &= \frac{1}{\sum^\text{in}_i w_i c_{i\alpha}^2}\left[\sum_j^{\text{par}}f_j c_{j\alpha} + \left(\frac{2}{c_s^2}\sum^\text{in}_j w_j c_{j\alpha}^2 - 1\right)\rho u_\alpha\right]
\end{split}
\end{equation}
After defining $C_\alpha = \sum^\text{in}_i w_i c_{i\alpha}^2$ and $D_\alpha = \left(\frac{2}{c_s^2}C_\alpha - 1\right)$, we can write the incoming populations in orders of Knudsen,
\begin{equation}
\begin{split}
    f_i^{(0)} &= f_i^{\text{eq},(0)} - \sum_\alpha \frac{w_i c_{i\alpha}}{C_\alpha} \sum_j^{\text{par}}f_j^{\text{eq},(0)} c_{j\alpha}\\
    f_i^{(1)} &= f_i^{\text{eq},(1)} - \sum_\alpha \frac{w_i c_{i\alpha}}{C_\alpha} \left(\sum_j^{\text{par}}f_j^{\text{eq},(1)} c_{j\alpha} + D_\alpha \rho^{(0)}u^{(1)}\right)\\
    f_i^{(2)} &= f_i^{\text{eq},(2)} + \frac{\tau}{\textit{Sh}}c_{i\alpha}\partial_\alpha f_q^{\text{eq},(1)} + (\text{momentum corrections})^{(2)}\\
    f_i^{(3)} &= f_i^{\text{eq},(3)} + \tau \partial_t f_i^{\text{eq}, (1)} - \frac{\tau}{\textit{Sh}}c_{i\alpha}\partial_\alpha f_i^{\text{eq},(2)}\\
    - \sum_\alpha &\frac{w_i c_{i\alpha}}{C_\alpha} \Biggl(\sum_j^{\text{par}}f_j^{(3)} c_{j\alpha} + D_\alpha\Bigl(\rho^{(0)}u^{(3)} + \rho^{(1)}u^{(2)} + \rho^{(2)}u^{(1)}\Bigr)\Biggr)\\
\end{split}
\end{equation}
